  \begin{block}{Theorem}
    \begin{itemize}
      \item<+->[]An SLList implements the Stack and (FIFO) Queue interfaces.
      \begin{itemize}
          \item<+->Stack operations $\fn{push}(x)$ and $\fn{pop}(x)$ each run in $O(1)$ time;
          \item<+->Queue operations $\fn{add}(x)$ and $\fn{remove}(x)$ each run in $O(1)$ time.
      \end{itemize}
    \end{itemize}
  \end{block}
